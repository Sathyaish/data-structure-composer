
ds VectorList {
    getByIndex <- 1
    updateNode! <- 1
    insertLast! <- 1
    deleteAtIndex! <- n
    deleteBetweenNodes! <- n
    deleteLast! <- 1
}

ds AugmentedRedBlackOrderStatisticTreeList[g] {
    getByIndex <- log(n)
    updateNode! <- log(n) + log(n) * g
    getNext <- 1
    getPrev <- 1
    insertLast! <- 1 + log(n) * g
    insertAfterNode! <- 1 + log(n) * g
    insertFirst! <- 1  + log(n) * g
    deleteNode! <- log(n) + log(n) * g
    deleteBetweenNodes! <- log(n) + log(n) * g
    twoSidedIndexRangeQuery[g] <- log(n) + log(n) * g
}

ds StackMemoizer[reduction] if reduction.idempotent {
    insertLast! <- 1
    deleteLast! <- 1
    reduce[reduction, zero] <- 1
}

ds Heap[heapOrdering] {
    unorderedEach <- n
    getFirstBy[heapOrdering] <- 1
    deleteNode! <- log(n)
    updateNode! <- log(n)
    insertAtIndex! <- log(n)
}

ds OrderedRedBlackTree[f, g] {
    unorderedEach <- n
    getNext <- 1
    getPrev <- 1
    getFirst <- 1
    getLast <- 1
    updateNode! <- 1 + g
    getFirstBy[f] <- 1
    getLastBy[f] <- 1
    getKthBy[f] <- log(n)
    insertAtIndex! <- log(n) + g
    getFirstNodeWithValue[f] <- log(n)
    deleteNode! <- 1 + g
    countBetweenBy[f] <- log(n)
    twoSidedValueRangeQuery[g] <- log(n) * g
    oneSidedValueRangeQuery[g] <- log(n) * g
}

ds RangeMinQueryLinearithmicArrayThing[reduction] if reduction.idempotent {
    twoSidedIndexRangeQuery[reduction] <- 1 + reduction
    insertLast! <- log(n) * reduction
    deleteLast! <- 1
}
